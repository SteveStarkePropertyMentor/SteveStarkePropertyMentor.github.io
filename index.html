    // Correct Formspree endpoint provided

    const USE_FORMSPREE = true;

    const FORMSPREE_ENDPOINT = "https://formspree.io/f/xzzkkjwd";


    const form = document.getElementById('contactForm');

    const ok = document.getElementById('contactSuccess');

    const err = document.getElementById('contactError');


    function validEmail(v){ return /.+@.+\..+/.test(v); }


    form.addEventListener('submit', async (e) => {

      e.preventDefault();

      ok.style.display = 'none';

      err.style.display = 'none';


      const data = Object.fromEntries(new FormData(form).entries());

      // honeypot trap

      if (data.website) { return; }

      if (!data.name || !validEmail(data.email) || !data.message){

        err.textContent = 'Please fill name, a valid email and your message.';

        err.style.display = 'block';

        return;

      }


      try{

        const res = await fetch(FORMSPREE_ENDPOINT, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({

            name: data.name,

            email: data.email,

            phone: data.phone || '',

            message: data.message

          })

        });

        if (res.ok){

          ok.style.display = 'block';

          form.reset();

          return;

        }

        // Formspree often returns 200/201; treat non-ok as error

        throw new Error('Bad status: ' + res.status);

      }catch(e){

        console.error(e);

        err.style.display = 'block';

      }

    });

  </script>
